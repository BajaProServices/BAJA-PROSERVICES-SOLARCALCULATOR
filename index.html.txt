        },

            historyTitle: "Recent Consumption (kWh)", tariff: "Rate Type:",

            gearTitle: "Système Photovoltaïque", panel: "Puissance du Panneau:", 
            resPanels: "Panneaux", resCost: "Investissement", resROI: "Retour (ROI)", 
            unit: "Période", roiYears: "Ans", btnClear: "EFFACER", btnQuote: "DEVIS WHATSAPP",

        }

    let currentLang = 'es';

    function setLang(lang) {
        currentLang = lang;
        document.querySelectorAll('.lang-bar button').forEach(b => b.classList.remove('active'));
        document.getElementById(`btn-${lang}`).classList.add('active');
        
        const t = translations[lang];
        document.getElementById('txt-client-title').innerText = t.clientTitle;
        document.getElementById('lbl-name').innerText = t.name;
        document.getElementById('lbl-service').innerText = t.service;
        document.getElementById('txt-history-title').firstChild.textContent = t.historyTitle;
        document.getElementById('lbl-tariff').innerText = t.tariff;
        document.getElementById('txt-gear-title').innerText = t.gearTitle;
        document.getElementById('lbl-panel').innerText = t.panel;
        document.getElementById('lbl-res-panels').innerText = t.resPanels;
        document.getElementById('lbl-res-cost').innerText = t.resCost;
        document.getElementById('lbl-res-roi').innerText = t.resROI;
        document.getElementById('btn-reset').innerText = t.btnClear;
        document.getElementById('btn-quote').innerText = t.btnQuote;
        document.getElementById('tagDAC').innerText = t.dacTag;
        
        // Update Select Options
        const sel = document.getElementById('tipoCliente');
        sel.options[0].text = t.domestica;
        sel.options[1].text = t.comercial;

        generarCampos();
    }

    function generarCampos() {
        const container = document.getElementById('inputsConsumo');
        const typeValue = document.getElementById('tipoCliente').value;
        const total = typeValue === "60" ? 6 : 12;
        container.innerHTML = '';
        for (let i = 1; i <= total; i++) {
            container.innerHTML += `<div><label>${translations[currentLang].unit} ${i}</label><input type="number" class="consumo-val" oninput="calcular()" placeholder="0"></div>`;
        }
    }

    function calcular() {
        const basePricePerWatt = 15.15;
        const tax = 1.16;
        const watts = parseFloat(document.getElementById('potenciaPanel').value);
        const type = document.getElementById('tipoCliente').value;
        
        let totalConsumption = 0;
        let count = 0;
        document.querySelectorAll('.consumo-val').forEach(i => {
            if(i.value > 0) { totalConsumption += parseFloat(i.value); count++; }
        });

        if (totalConsumption > 0) {
            const avgBiMonthly = (type === "60") ? (totalConsumption / (count || 1)) : (totalConsumption / (count || 1) * 2);
            
            let kwhPrice = 0;
            let isDAC = false;

            if (type === "60") {
                if (avgBiMonthly > 500) { kwhPrice = 6.50; isDAC = true; } 
                else if (avgBiMonthly > 280) { kwhPrice = 3.51; }
                else { kwhPrice = 1.10; }
            } else {
                kwhPrice = 6.20; 
            }

            document.getElementById('tagDAC').style.display = isDAC ? "inline" : "none";

            const panelsNeeded = Math.ceil(((totalConsumption / 365) / 5.9 / 0.8) * 1000 / watts);
            const totalInvestment = (panelsNeeded * watts * basePricePerWatt) * tax;
            
            const annualKwh = (type === "60") ? (totalConsumption * (6/(count||1))) : (totalConsumption * (12/(count||1)));
            const annualSavings = (annualKwh * kwhPrice) * tax;
            const roi = (totalInvestment / annualSavings).toFixed(1);

            document.getElementById('resPaneles').innerText = panelsNeeded;
            document.getElementById('resPrecio').innerText = '$' + totalInvestment.toLocaleString('es-MX', {maximumFractionDigits: 0});
            document.getElementById('resROI').innerText = `${roi} ${translations[currentLang].roiYears}`;
            
            const infoMsg = currentLang === 'es' ? `Ahorro anual est.: $${annualSavings.toLocaleString()} MXN` : 
                           currentLang === 'en' ? `Est. Annual Savings: $${annualSavings.toLocaleString()} MXN` :
                           `Économies annuelles est.: $${annualSavings.toLocaleString()} MXN`;
            document.getElementById('roi-msg').innerText = infoMsg;
        }
    }

    function limpiarDatos() {
        if(confirm("¿Limpiar datos? / Clear data?")) location.reload();
    }

    function enviarInformacion() {
        const name = document.getElementById('nombre').value || "Client";
        const panels = document.getElementById('resPaneles').innerText;
        const cost = document.getElementById('resPrecio').innerText;
        const roi = document.getElementById('resROI').innerText;
        const msg = `*SOLAR QUOTE - BAJA PRO*%0A👤 Name: ${name}%0A☀️ Panels: ${panels}%0A💰 Investment: ${cost}%0A⏳ ROI: ${roi}`;
        window.open(`https://wa.me/526122198325?text=${msg}`, '_blank');
    }
